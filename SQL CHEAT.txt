--create database

CREATE DATABASE <NAMA DATABASE>;

--CREATE TABLE

CREATE TABLE <NAMA TABLE> (
	userid int,
	first_name VARCHAR(100),
	last_name VARCHAR(100),
	email VARCHAR(225)
);

--MENAMBAHKAN COLOUMN DARI TABLE YANG ADA

ALTER TABLE <NAMA TABLE> ADD <NAMA COLUMN> <JENIS DATA> (<JUMLAH KARAKTER>);
ex:
ALTER TABLE users ADD encrypted_password VARCHAR (1000);

--HAPUS COLUMN YANG ADA 

ALTER TABLE <NAMA TABLE> DROP COLOUMN <NAMA COLUMN>
ex:
ALTER TABLE users DROP COLUMN email;

--HAPUS TABLE

DROP <NAMA TABLE>;

--HAPUS DATABASE

DROP <NAMA DATABASE>;

--MEMASUKKAN VALUE DARI TABLE YANG SUDAH DI BUAT

INSERT INTO <NAMA TABLE> (<NAMA COLOUMN>, <NAMA COLOUMN>, <NAMA COLOUMN>)
VALUE (<VALUE>, <VALUE>, <VALUE>);
ex:
INSERT INTO users (movie_id, title, description, price)
VALUE (1, 'Gattaca', 'Movie or documentary?', 4.99);

--mengubah value yang ada di table

UPDATE <NAMA TABLE> SET <NAMA COLOUMN> = <VALUE BARU> WHERE <KONDISI> = <NILAI KONDISI>;
ex:
UPDATE movie SET price = 0.99 WHERE tittle = 'jaws';

--MENGHAPUS DATA 

DELETE FROM <NAMA TABLE> WHERE <KONDISI> = <NILAI KONDISI>;
ex:
DELETE FROM movie  WHERE title = 'Star Wars';
